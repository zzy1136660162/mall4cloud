# 选品功能前端数据动态方案

## 一、需求分析

### 1.1 业务流程
```
用户选品流程：
1. 浏览选品列表 → 2. 查看商品详情 → 3. 申请选品 → 4. 填写申请信息 → 5. 提交申请
```

### 1.2 数据来源
- **商品列表/详情** → 通过 `mall4cloud-product` 微服务获取
- **选品申请** → 通过 `mall4cloud-platform` 微服务提交

---

## 二、数据接口设计

### 2.1 商品列表接口（product微服务）

**来源**：`mall4cloud-product` 的 `SpuController` 或 `SpuService`

**接口路径**：`GET /product/spu/page` 或 `GET /product/selection/list`

**请求参数**：
```json
{
  "pageNum": 1,
  "pageSize": 10,
  "isSelection": 1,           // 选品标识
  "isHot": null,               // 热销商品
  "isNew": null,               // 新品
  "isRecommend": null,          // 推荐
  "isSelectionBest": null,      // 精选
  "minCommissionRate": null,     // 最低佣金比例
  "minRating": null,             // 最低评分
  "orderBy": "totalSales",      // 排序字段
}
```

**响应数据**：
```json
{
  "code": 200,
  "msg": "success",
  "data": {
    "list": [
      {
        "spuId": 1,
        "name": "商品名称",
        "mainImgUrl": "https://xxx.jpg",
        "priceFee": 9999,
        "commissionRate": 10.00,
        "commissionAmount": 999.90,
        "isHot": 1,
        "isNew": 0,
        "totalSales": 1000,
        "monthSales": 100,
        "rating": 4.5
      }
    ],
    "total": 100,
    "pageNum": 1,
    "pageSize": 10
  }
}
```

### 3.2 API封装

**文件**：`src/api/selection.js`

```javascript
const { get, post } = require('./request')

// API基础路径
const API_PRODUCT = '/mall4cloud_product/product/spu'  // product微服务
const API_PLATFORM = '/mall4cloud_platform'           // platform微服务

module.exports = {
  /**
   * 获取选品商品列表
   * @param {Object} params 查询参数
   */
  getSelectionProducts(params = {}) {
    return get(`${API_PRODUCT}/page`, {
      isSelection: 1,
      ...params
    })
  },

  /**
   * 获取选品商品详情
   * @param {String|Number} spuId 商品ID
   */
  getSelectionDetail(spuId) {
    return get(`${API_PRODUCT}/detail/${spuId}`)
  },

  /**
   * 获取选品商品列表（新接口）
   * @param {Object} params 查询参数
   */
  getSelectionList(params = {}) {
    return get(`${API_PRODUCT}/selection/list`, params)
  },

  /**
   * 提交选品申请
   * @param {Object} data 申请数据
   */
  submitSelectionApply(data) {
    return post(`${API_PLATFORM}/app/selection/apply`, data)
  },

  /**
   * 获取我的选品申请列表
   * @param {Object} params 分页参数
   */
  getMySelectionApplies(params = {}) {
    return get(`${API_PLATFORM}/app/selection/my-applies`, params)
  }
}
```

### 3.3 选品列表页数据获取

**文件**：`src/pages/selection/list.vue`

```javascript
// script部分
import selectionApi from '@/api/selection'

export default {
  data() {
    return {
      // 筛选参数
      filters: {
        category: '',           // 分类
        sortBy: 'totalSales',  // 排序字段
        minCommission: null,     // 最低佣金
        order: 'desc'          // 排序顺序
      },
      // 列表数据
      products: [],
      // 分页
      pageNum: 1,
      pageSize: 10,
      total: 0,
      loading: false
    }
  },
  
  onLoad() {
    this.loadProducts()
  },
  
  // 方法
  methods: {
    async loadProducts() {
      this.loading = true
      try {
        const res = await selectionApi.getSelectionProducts({
          pageNum: this.pageNum,
          pageSize: this.pageSize,
          ...this.filters
        })
        
        if (res.code === 200) {
          this.products = res.data.list
          this.total = res.data.total
        }
      } catch (error) {
        uni.showToast({ title: '加载失败' })
      } finally {
        this.loading = false
      }
    },
    
    // 筛选
    onFilter() {
      this.pageNum = 1
      this.loadProducts()
    },
    
    // 上拉加载
    onReachBottom() {
      if (this.products.length < this.total) {
        this.pageNum++
        this.loadProducts()
      }
    },
    
    // 跳转详情
    goToDetail(spuId) {
      uni.navigateTo({
        url: `/pages/selection/detail?spuId=${spuId}`
      })
    }
  }
}
```

### 3.4 选品详情页数据获取

**文件**：`src/pages/selection/detail.vue`

```javascript
import selectionApi from '@/api/selection'

export default {
  data() {
    return {
      spuId: null,
      product: {},
      loading: false
    }
  },
  
  onLoad(options) {
    this.spuId = options.spuId
    this.loadDetail()
  },
  
  methods: {
    async loadDetail() {
      this.loading = true
      try {
        const res = await selectionApi.getSelectionDetail(this.spuId)
        if (res.code === 200) {
          this.product = res.data
        }
      } catch (error) {
        uni.showToast({ title: '加载失败' })
      } finally {
        this.loading = false
      }
    }
  }
}
```

### 3.5 申请选品数据提交

**文件**：`src/pages/selection/apply.vue`

```javascript
import selectionApi from '@/api/selection'

export default {
  data() {
    return {
      form: {
        spuId: null,
        contactPhone: '',
        deliveryAddress: ''
      },
      submitting: false
    }
  },
  
  methods: {
    async submitApply() {
      // 表单验证
      if (!this.form.contactPhone) {
        return uni.showToast({ title: '请填写手机号' })
      }
      if (!this.form.deliveryAddress) {
        return uni.showToast({ title: '请填写地址' })
      }
      
      this.submitting = true
      try {
        const res = await selectionApi.submitSelectionApply(this.form)
        if (res.code === 200) {
          uni.showToast({ title: '申请成功' })
          uni.navigateBack()
        }
      } catch (error) {
        uni.showToast({ title: '申请失败' })
      } finally {
        this.submitting = false
      }
    }
  }
}
```

## 六、注意事项

1. **跨服务调用**：选品列表/详情走 product 微服务，申请走 platform 微服务
2. **数据缓存**：商品详情可做本地缓存
3. **下拉刷新**：列表页支持下拉刷新和上拉加载
4. **表单验证**：申请页做好手机号、地址校验
5. **错误处理**：做好网络异常处理和用户提示
