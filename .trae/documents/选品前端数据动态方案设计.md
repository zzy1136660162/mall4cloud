# 选品功能前端数据动态方案

## 一、需求分析

### 1.1 业务流程
```
用户选品流程：
1. 浏览选品列表 → 2. 查看商品详情 → 3. 申请选品 → 4. 填写申请信息 → 5. 提交申请
```

### 1.2 数据来源
- **商品列表/详情** → 通过 `mall4cloud-product` 微服务获取
- **选品申请** → 通过 `mall4cloud-platform` 微服务提交
- **商品分类** → 通过现有的商品分类接口获取（使用 `/ua/category/category_list`）

### 1.3 项目现状
- ✅ 已存在选品页面模板：`src/pages/sample/selection/selection.vue`（使用mock数据）
- ✅ 已存在申请页面模板：`src/pages/sample/apply/apply.vue`
- ✅ 已存在商品详情页面：`src/pages/sample/product/product.vue`
- ✅ 后端已实现选品申请接口：`SelectionApplyController`
- ✅ 后端已支持 `isSelection` 字段筛选：`SpuPageSearchDTO`

### 3.2 API封装

**文件**：`src/utils/api/selection.js`（新建）

```javascript
import http from '@/utils/http.js'

const API_PRODUCT = '/mall4cloud_product/ua'
const API_PLATFORM = '/mall4cloud_platform'

export default {
  /**
   * 获取商品分类列表（使用现有接口）
   * @param {Number} parentId 父分类ID（0获取一级分类）
   * @param {Number} shopId 店铺ID（0表示平台分类）
   */
  getCategories(parentId = 0, shopId = 0) {
    return http.request({
      url: `${API_PRODUCT}/category/category_list`,
      method: 'GET',
      data: { parentId, shopId }
    })
  },

  /**
   * 获取选品商品列表
   * @param {Object} params 查询参数
   */
  getSelectionProducts(params = {}) {
    return http.request({
      url: `${API_PRODUCT}/spu/page`,
      method: 'GET',
      data: {
        isSelection: 1,
        ...params
      }
    })
  },

  /**
   * 获取选品商品详情
   * @param {String|Number} spuId 商品ID
   */
  getSelectionDetail(spuId) {
    return http.request({
      url: `${API_PRODUCT}/spu/detail/${spuId}`,
      method: 'GET'
    })
  },

  /**
   * 提交选品申请
   * @param {Object} data 申请数据
   */
  submitSelectionApply(data) {
    return http.request({
      url: `${API_PLATFORM}/app/selection/apply`,
      method: 'POST',
      data
    })
  },

  /**
   * 获取我的选品申请列表
   * @param {Object} params 分页参数
   */
  getMySelectionApplies(params = {}) {
    return http.request({
      url: `${API_PLATFORM}/app/selection/my-applies`,
      method: 'GET',
      data: params
    })
  },

  /**
   * 获取申请详情
   * @param {String|Number} applyId 申请ID
   */
  getApplyDetail(applyId) {
    return http.request({
      url: `${API_PLATFORM}/app/selection/detail/${applyId}`,
      method: 'GET'
    })
  }
}
```

### 3.3 选品列表页数据获取（修改现有页面）

**文件**：`src/pages/sample/selection/selection.vue`

```javascript
import { ref, reactive, onMounted } from 'vue'
import selectionApi from '@/utils/api/selection.js'

const currentCategory = ref(0)
const currentFilter = ref('all')
const products = ref([])
const page = ref(1)
const pageSize = 10
const loading = ref(false)
const hasMore = ref(true)
const showFilterPanel = ref(false)
const categories = ref([])
const currentFilters = reactive({
  commission: null,
  priceMin: null,
  priceMax: null
})

onMounted(() => {
  loadCategories()
  loadProducts()
})

async function loadCategories() {
  try {
    const res = await selectionApi.getCategories(0, 0)
    if (res && res.length) {
      categories.value = res
    }
  } catch (error) {
    console.error('加载分类失败', error)
  }
}

async function loadProducts() {
  if (loading.value) return
  loading.value = true

  try {
    const params = {
      pageNum: page.value,
      pageSize: pageSize,
      categoryId: currentCategory.value || null
    }

    // 根据筛选条件添加参数
    if (currentFilters.commission) {
      params.minCommissionRate = parseFloat(currentFilters.commission)
    }
    if (currentFilters.priceMin) {
      params.minPrice = currentFilters.priceMin * 100
    }
    if (currentFilters.priceMax) {
      params.maxPrice = currentFilters.priceMax * 100
    }

    // 排序处理
    if (currentFilter.value === 'hot') {
      params.saleNumSort = 0
    } else if (currentFilter.value === 'new') {
      params.createTimeSort = 0
    } else if (currentFilter.value === 'commission') {
      params.commissionSort = 0
    }

    const res = await selectionApi.getSelectionProducts(params)

    if (res && res.list) {
      if (page.value === 1) {
        products.value = res.list
      } else {
        products.value = [...products.value, ...res.list]
      }
      hasMore.value = products.value.length < res.total
    }
  } catch (error) {
    uni.showToast({ title: '加载失败', icon: 'none' })
  } finally {
    loading.value = false
  }
}

function switchCategory(categoryId) {
  currentCategory.value = categoryId
  page.value = 1
  loadProducts()
}

function switchFilter(filter) {
  currentFilter.value = filter
  page.value = 1
  loadProducts()
}

function toggleFilterPanel() {
  showFilterPanel.value = !showFilterPanel.value
}

function selectSubFilter(type, value) {
  if (type === 'commission') {
    currentFilters.commission = value === 'all' ? null : value
  } else if (type === 'price') {
    if (value === 'all') {
      currentFilters.priceMin = null
      currentFilters.priceMax = null
    } else if (value === '50') {
      currentFilters.priceMax = 50
      currentFilters.priceMin = null
    } else if (value === '500') {
      currentFilters.priceMin = 500
      currentFilters.priceMax = null
    } else if (value.includes('-')) {
      const [min, max] = value.split('-')
      currentFilters.priceMin = parseInt(min)
      currentFilters.priceMax = parseInt(max)
    }
  }
  page.value = 1
  loadProducts()
}

function goToProductDetail(spuId) {
  uni.navigateTo({
    url: `/pages/sample/product/product?spuId=${spuId}`
  })
}

function applySample(spuId) {
  uni.navigateTo({
    url: `/pages/sample/apply/apply?spuId=${spuId}`
  })
}

function onPullDownRefresh() {
  page.value = 1
  loadProducts().finally(() => {
    uni.stopPullDownRefresh()
  })
}

function onReachBottom() {
  if (hasMore.value && !loading.value) {
    page.value++
    loadProducts()
  }
}
```

### 3.5 申请选品页数据提交

**文件**：`src/pages/sample/apply/apply.vue`（已有模板，需完善）

```javascript
import selectionApi from '@/utils/api/selection.js'

export default {
  data() {
    return {
      spuId: null,
      form: {
        contactName: '',
        contactPhone: '',
        deliveryAddress: '',
        remark: ''
      },
      submitting: false
    }
  },

  onLoad(options) {
    this.spuId = options.spuId
    this.loadDefaultAddress()
  },

  methods: {
    async loadDefaultAddress() {
      const address = uni.getStorageSync('cloudDeliveryAddrInfo')
      if (address) {
        this.form.contactName = address.consignee
        this.form.contactPhone = address.mobile
        this.form.deliveryAddress = `${address.province}${address.city}${address.district}${address.addrDetails}`
      }
    },

    selectAddress() {
      uni.navigateTo({
        url: '/pages/address-list/address-list?type=select',
        events: {
          selectAddress: (address) => {
            this.form.contactName = address.consignee
            this.form.contactPhone = address.mobile
            this.form.deliveryAddress = `${address.province}${address.city}${address.district}${address.addrDetails}`
          }
        }
      })
    },

    validateForm() {
      if (!this.form.contactName) {
        uni.showToast({ title: '请填写收件人', icon: 'none' })
        return false
      }
      if (!this.form.contactPhone) {
        uni.showToast({ title: '请填写手机号', icon: 'none' })
        return false
      }
      if (!/^1[3-9]\d{9}$/.test(this.form.contactPhone)) {
        uni.showToast({ title: '手机号格式不正确', icon: 'none' })
        return false
      }
      if (!this.form.deliveryAddress) {
        uni.showToast({ title: '请填写收货地址', icon: 'none' })
        return false
      }
      return true
    },

    async submitApply() {
      if (!this.validateForm()) return
      if (this.submitting) return

      this.submitting = true
      try {
        const res = await selectionApi.submitSelectionApply({
          spuId: this.spuId,
          ...this.form
        })

        if (res && res.applyId) {
          uni.showToast({ title: '申请成功', icon: 'success' })
          setTimeout(() => {
            uni.navigateBack()
          }, 1500)
        }
      } catch (error) {
        uni.showToast({ title: error.msg || '申请失败', icon: 'none' })
      } finally {
        this.submitting = false
      }
    }
  }
}
```

### 6.1 实施步骤

   - 新建 `src/utils/api/selection.js`
   - 封装所有选品相关接口

2. **修改选品列表页**
   - 修改 `src/pages/sample/selection/selection.vue`
   - 将mock数据替换为API调用
   - 使用现有分类接口获取分类数据
   - 实现筛选、排序、分页功能

3. **完善商品详情页**
   - 新建或完善 `src/pages/sample/product/product.vue`
   - 调用商品详情接口
   - 展示完整商品信息

4. **完善申请选品页**
   - 完善 `src/pages/sample/apply/apply.vue`
   - 实现地址选择功能
   - 表单验证和提交

5. **新建我的申请列表页**
   - 新建 `src/pages/sample/list/list.vue`
   - 实现申请列表查询

### 6.2 注意事项

1. **跨服务调用**
   - 商品列表/详情：调用 `mall4cloud-product` 微服务
   - 选品申请：调用 `mall4cloud-platform` 微服务
   - 需要正确配置网关路由

2. **分类接口使用**
   - 使用现有接口 `/ua/category/category_list`
   - parentId=0 获取一级分类
   - shopId=0 获取平台分类

3. **价格单位转换**
   - 前端显示价格单位：元
   - 后端存储价格单位：分
   - 请求时需要 `price * 100`
   - 响应时需要 `price / 100`

4. **数据缓存**
   - 商品详情可做本地缓存（减少频繁请求）
   - 分类数据可做本地缓存（变化较少）

5. **下拉刷新和上拉加载**
   - 列表页需要支持下拉刷新
   - 列表页需要支持上拉加载更多
   - 需要显示加载状态和"没有更多"提示

6. **表单验证**
   - 申请页需要验证手机号格式
   - 需要验证必填字段
   - 地址信息建议从已有地址簿选择

7. **错误处理**
   - 做好网络异常处理
   - 做好接口返回错误码处理
   - 需要友好的用户提示
